-- TABELA 1: Categoria
CREATE TABLE Categoria (
    CAT_ID SERIAL PRIMARY KEY,
    CAT_NOME_PROD VARCHAR(150),
    CAT_ESPECIFICACAO VARCHAR(130)
);

-- TABELA 2: Produto
CREATE TABLE Produto (
    PROD_ID SERIAL PRIMARY KEY,
    PROD_NOME VARCHAR(100),
    PROD_DESCRICAO VARCHAR(150),
    PROD_INFORMACOES_ADICIONAIS VARCHAR(150),
    PROD_QUANT INTEGER,
    CAT_ID INTEGER,
    FOREIGN KEY (CAT_ID) REFERENCES Categoria(CAT_ID)
);

-- TABELA 3: Donatario
CREATE TABLE Donatario (
    DON_ID SERIAL PRIMARY KEY,
    DON_NOME VARCHAR(100) NOT NULL,
    DON_DATA_NASC DATE,
    DON_RUA VARCHAR(80),
    DON_BAIRRO VARCHAR(80),
    DON_CIDADE VARCHAR(80),
    DON_TELEFONE VARCHAR(12),
    DON_CEP CHAR(8),
    DON_UF CHAR(2),
    DON_EMAIL VARCHAR(100),
    DON_SEXO VARCHAR(20) CHECK (DON_SEXO IN ('Masculino', 'Feminino', 'Não Informado')),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- INSERÇÃO DA TABELA VOLUNTARIO (Com dados cadastrais)
CREATE TABLE Voluntario (
    VOL_ID SERIAL PRIMARY KEY,
    VOL_NOME VARCHAR(100) NOT NULL,
    VOL_TELEFONE VARCHAR(12),
    VOL_EMAIL VARCHAR(100),
    VOL_CIDADE VARCHAR(80),
    VOL_BAIRRO VARCHAR(80), -- Corrigido de VOL_BARRO para VOL_BAIRRO
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
    -- Outros campos cadastrais do voluntário podem ser adicionados aqui (ex: CPF, Data de Nascimento, etc.)
);

-- TABELA 4: Login (Apenas dados de autenticação)
CREATE TABLE Login (
    VOL_ID INTEGER PRIMARY KEY, -- Agora é a chave primária e referência ao Voluntário
    login VARCHAR(50) UNIQUE NOT NULL,
    senha VARCHAR(255) NOT NULL, -- Aumentado para suportar hashes maiores
    nive_acesso VARCHAR(20) CHECK (nive_acesso IN ('Administrador', 'Logistica', 'Operacional')),
    status VARCHAR(10) CHECK (status IN ('Ativo', 'Inativo')),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (VOL_ID) REFERENCES Voluntario(VOL_ID)
);

-- TABELA 5: Doadores
CREATE TABLE Doadores (
    DOA_ID SERIAL PRIMARY KEY,
    DOA_NOME VARCHAR(200) NOT NULL,
    DOA_TELEFONE VARCHAR(12),
    DOA_EMAIL VARCHAR(100),
    DOA_CEP CHAR(8),
    DOA_UF CHAR(2),
    DOA_CIDADE VARCHAR(150),
    DOA_BAIRRO VARCHAR(150),
    DOA_RUA VARCHAR(150),
    DOA_CPF CHAR(11), -- CPF tem 11 dígitos
    DOA_DATANASC DATE,
    DOA_SEXO VARCHAR(10) CHECK (DOA_SEXO IN ('Masculino', 'Feminino', 'Jurídico')),
    DOA_SITE VARCHAR(255),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Alteração de colunas removida pois já foi definido corretamente acima

-- TABELA 6: Doados
-- FK VOL_ID AGORA REFERENCIA Voluntario
CREATE TABLE Doados (
    DOA_ID SERIAL PRIMARY KEY,
    DOA_DATA_AQUISICAO DATE NOT NULL,
    DOA_TIPO_DOACAO VARCHAR(20) CHECK (DOA_TIPO_DOACAO IN ('Itens', 'Dinheiro', 'Misto')),
    VOL_ID INTEGER NOT NULL,
    DON_ID INTEGER,
    OBS_DOADO VARCHAR(500),
    VALOR_DOACAO NUMERIC(10, 2) DEFAULT 0.00,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (VOL_ID) REFERENCES Voluntario(VOL_ID),
    FOREIGN KEY (DON_ID) REFERENCES Donatario(DON_ID)
);

-- TABELA 7: Campanha
-- FK VOL_ID AGORA REFERENCIA Voluntario
CREATE TABLE Campanha (
    CAM_ID SERIAL PRIMARY KEY,
    CAM_DATA_INI DATE NOT NULL,
    CAM_DATA_FIM DATE NOT NULL,
    CAM_DESC VARCHAR(500) NOT NULL,
    CAM_META_ARRECADACAO NUMERIC(10, 2) NOT NULL,
    CAM_VALOR_ARRECADADO NUMERIC(10, 2) DEFAULT 0.00,
    VOL_ID INTEGER NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (VOL_ID) REFERENCES Voluntario(VOL_ID),
    CHECK (CAM_DATA_FIM >= CAM_DATA_INI)
);

-- TABELA 8: Agendar_Doados
-- FK VOL_ID AGORA REFERENCIA Voluntario
CREATE TABLE Agendar_Doados (
    AG_ID SERIAL PRIMARY KEY,
    AG_DATA DATE NOT NULL,
    AG_HORA TIME NOT NULL,
    AG_OBS VARCHAR(500),
    DOA_ID INTEGER NOT NULL,
    VOL_ID INTEGER NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (DOA_ID) REFERENCES Doados(DOA_ID),
    FOREIGN KEY (VOL_ID) REFERENCES Voluntario(VOL_ID)
);

-- TABELA 9: Campanha_Responsavel
-- FK VOL_ID AGORA REFERENCIA Voluntario
CREATE TABLE Campanha_Responsavel (
    CAMP_ID INTEGER,
    VOL_ID INTEGER,
    DATA_INICIO DATE NOT NULL,
    DATA_FIM DATE NOT NULL,
    OBS_TEXTO VARCHAR(500),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (CAMP_ID, VOL_ID),
    FOREIGN KEY (CAMP_ID) REFERENCES Campanha(CAM_ID),
    FOREIGN KEY (VOL_ID) REFERENCES Voluntario(VOL_ID),
    CHECK (DATA_FIM >= DATA_INICIO)
);

-- TABELA 10: Agendar_Retirada
-- FK VOL_ID AGORA REFERENCIA Voluntario
CREATE TABLE Agendar_Retirada (
    AGENDA_ID SERIAL PRIMARY KEY,
    DATA_RETIRO DATE NOT NULL,
    HORA_RETIRO TIME NOT NULL,
    OBS_RETIRO VARCHAR(500),
    VOL_ID INTEGER NOT NULL,
    DOA_ID INTEGER NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (VOL_ID) REFERENCES Voluntario(VOL_ID),
    FOREIGN KEY (DOA_ID) REFERENCES Doados(DOA_ID)
);

-- TABELA 11: Notificacao
-- FK VOL_ID AGORA REFERENCIA Voluntario
CREATE TABLE Notificacao (
    NOT_ID SERIAL PRIMARY KEY,
    NOT_TEXTO VARCHAR(500) NOT NULL,
    NOT_DATA DATE NOT NULL,
    NOT_HORA TIME NOT NULL,
    VOL_ID INTEGER NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (VOL_ID) REFERENCES Voluntario(VOL_ID)
);

-- TABELA 12: Notificacao_Donatario
CREATE TABLE Notificacao_Donatario (
    DON_ID INTEGER,
    NOT_ID INTEGER,
    PRIMARY KEY (DON_ID, NOT_ID),
    FOREIGN KEY (DON_ID) REFERENCES Donatario(DON_ID),
    FOREIGN KEY (NOT_ID) REFERENCES Notificacao(NOT_ID)
);

-- TABELA 13: DoadosProduto
CREATE TABLE DoadosProduto (
    DOA_ID INTEGER,
    PROD_ID INTEGER,
    DP_QTDE NUMERIC,
    PRIMARY KEY (DOA_ID, PROD_ID),
    FOREIGN KEY (DOA_ID) REFERENCES Doados(DOA_ID),
    FOREIGN KEY (PROD_ID) REFERENCES Produto(PROD_ID)
);

-- TABELA 14: Higienizacao_Roupas
-- FK VOL_ID AGORA REFERENCIA Voluntario
CREATE TABLE Higienizacao_Roupas (
    HIG_ID SERIAL PRIMARY KEY,
    HIG_DATA_AGENDADA DATE NOT NULL,
    HIG_DESCRICAO_ROUPA VARCHAR(200) NOT NULL,
    VOL_ID INTEGER NOT NULL,
    HIG_LOCAL VARCHAR(100),
    HIG_HORA TIME NOT NULL,
    HIG_VALORPAGO NUMERIC(10, 2) DEFAULT 0.00,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (VOL_ID) REFERENCES Voluntario(VOL_ID)
);

-- TABELA 15: Doacao
-- FK VOL_ID AGORA REFERENCIA Voluntario
CREATE TABLE Doacao (
    DOACAO_ID SERIAL PRIMARY KEY,
    VOL_ID INTEGER NOT NULL,
    DATA_DOACAO DATE NOT NULL,
    OBS_DOACAO VARCHAR(500),
    VALOR_DOADO NUMERIC(10, 2) DEFAULT 0.00,
    DOA_ID INTEGER NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (VOL_ID) REFERENCES Voluntario(VOL_ID),
    FOREIGN KEY (DOA_ID) REFERENCES Doadores(DOA_ID)
);

-- TABELA 16: Verificar_Necessidade
-- FK VOL_ID AGORA REFERENCIA Voluntario
-- CORRIGIDO: FK DON_ID deveria referenciar Donatario, não Doadores
CREATE TABLE Verificar_Necessidade (
    VER_ID SERIAL PRIMARY KEY,
    VER_DATA DATE NOT NULL,
    VER_OBS VARCHAR(500),
    VER_RESULTADO VARCHAR(100),
    VOL_ID INTEGER NOT NULL,
    DON_ID INTEGER NOT NULL, -- Corrigido de DOA_ID para DON_ID
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (VOL_ID) REFERENCES Voluntario(VOL_ID),
    FOREIGN KEY (DON_ID) REFERENCES Donatario(DON_ID) -- Corrigido para referenciar Donatario
);

-- TABELA 17: Acerto_Estoque
-- FK VOL_ID AGORA REFERENCIA Voluntario
CREATE TABLE Acerto_Estoque (
    AC_ID SERIAL PRIMARY KEY,
    AC_DATA DATE NOT NULL,
    AC_MOTIVO VARCHAR(30) CHECK (AC_MOTIVO IN ('Perda', 'Ajuste', 'Roubo', 'Doação', 'Vencimento', 'Outro')),
    AC_OBS VARCHAR(500),
    AC_TIPO VARCHAR(40) CHECK (AC_TIPO IN ('Entrada', 'Saída')),
    AC_QUANTIDADE NUMERIC(10, 2) NOT NULL,
    VOL_ID INTEGER NOT NULL,
    PROD_ID INTEGER NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (VOL_ID) REFERENCES Voluntario(VOL_ID),
    FOREIGN KEY (PROD_ID) REFERENCES Produto(PROD_ID)
);

-- TABELA 18: Compra
-- FK VOL_ID AGORA REFERENCIA Voluntario
CREATE TABLE Compra (
    COMP_ID SERIAL PRIMARY KEY,
    COMP_DATA_COMPRA DATE NOT NULL,
    COMP_DESC VARCHAR(200),
    VOL_ID INTEGER NOT NULL,
    COM_VALOR_TOTAL NUMERIC(10, 2) NOT NULL,
    COM_FORNECEDOR VARCHAR(50),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (VOL_ID) REFERENCES Voluntario(VOL_ID)
);

-- TABELA 19: DoacaoProduto
CREATE TABLE DoacaoProduto (
    DOACAO_ID INTEGER,
    PROD_ID INTEGER,
    QUANTIDADE_DOADA NUMERIC(10, 2) NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (DOACAO_ID, PROD_ID),
    FOREIGN KEY (DOACAO_ID) REFERENCES Doacao(DOACAO_ID),
    FOREIGN KEY (PROD_ID) REFERENCES Produto(PROD_ID)
);

-- TABELA 20: Compra_Produto
CREATE TABLE Compra_Produto (
    COMP_ID INTEGER,
    PROD_ID INTEGER,
    QTDE NUMERIC(10, 2) NOT NULL,
    VALOR_UNITARIO NUMERIC(10, 2) NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (COMP_ID, PROD_ID),
    FOREIGN KEY (COMP_ID) REFERENCES Compra(COMP_ID),
    FOREIGN KEY (PROD_ID) REFERENCES Produto(PROD_ID)
);

-- TABELA 21: Parametrizacao

    P_EMAIL VARCHAR(100),
    P_SITE VARCHAR(255),
    P_TELEFONE VARCHAR(12),
    P_CONTATO VARCHAR(15),
    P_LOGOTIPO_GRANDE VARCHAR(255),
    P_LOGOTIPO_PEQUENO VARCHAR(255),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
